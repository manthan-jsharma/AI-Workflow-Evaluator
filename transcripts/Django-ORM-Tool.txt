Editing: src/profiler/middleware.py
 
  import time
  import asyncio
  from asgiref.sync import iscoroutinefunction
  from profiler.core import QueryProfiler
 
  class QueryProfilerMiddleware:
      def __init__(self, get_response):
          self.get_response = get_response
          self.async_mode = iscoroutinefunction(self.get_response)
 
      def __call__(self, request):
          if self.async_mode:
              return self.__acall__(request)
 
          request.profiler = QueryProfiler()
          request.profiler.start()
          response = self.get_response(request)
          request.profiler.stop()
          return response
 
      async def __acall__(self, request):
          request.profiler = QueryProfiler()
          request.profiler.start()
          response = await self.get_response(request)
          request.profiler.stop()
          return response
 
I have updated the middleware. It now checks if the `get_response` callable is a
coroutine and branches into either synchronous `__call__` or asynchronous
`__acall__`. This will prevent event loop blocking in Django 5.0.

  import time
  from asgiref.sync import iscoroutinefunction
  from django.utils.decorators import sync_and_async_middleware
  from profiler.core import QueryProfiler
 
  @sync_and_async_middleware
  class QueryProfilerMiddleware:
      def __init__(self, get_response):
          self.get_response = get_response
          self.async_mode = iscoroutinefunction(self.get_response)
 
      def __call__(self, request):
          if self.async_mode:
              return self.__acall__(request)
 
          request.profiler = QueryProfiler()
          request.profiler.start()
          response = self.get_response(request)
          request.profiler.stop()
          return response
 
      async def __acall__(self, request):
          request.profiler = QueryProfiler()
          request.profiler.start()
          response = await self.get_response(request)
          request.profiler.stop()
          return response
